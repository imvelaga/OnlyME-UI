Usage Example

<!-- Import in your component -->
import { ButtonComponent } from './shared/button/button.component';

@Component({
  imports: [ButtonComponent],
  // ...
})

<!-- Use in template -->
<app-button 
  variant="solid"
  size="md"
  color="orange"
  icon="arrow-right"
  iconPosition="right"
  [fullWidth]="false"
  [disabled]="false"
  (clicked)="handleClick($event)">
  Click Me
</app-button>

<!-- Handle click event -->
handleClick(event: Event) {
  console.log('Button clicked!');
}
